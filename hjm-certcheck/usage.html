# 使用

hjm-certcheck 安装完成后将注册为 systemd 服务.

# 启用(开机自启)

systemctl enable hjm-certcheck.service

# 启动服务

systemctl start hjm-certcheck.service

# 停止服务

systemctl stop hjm-certcheck.service

# 重启服务

systemctl restart hjm-certcheck.service

# 查看服务状态

systemctl status hjm-certcheck.service --no-pager

# 查看服务日志

journalctl -u hjm-certcheck.service -f

# 停用服务

systemctl disable hjm-certcheck.service

# 删除服务

systemctl stop hjm-certcheck.service
systemctl disable hjm-certcheck.service
rm /etc/systemd/system/hjm-certcheck.service
systemctl daemon-reload
